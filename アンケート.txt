<script>
const DEFAULT_PASSWORD = 'admin1234';

function getData(key, fallback = []) {
  try { return JSON.parse(localStorage.getItem(key)) ?? fallback; } catch { return fallback; }
}
function setData(key, val) { localStorage.setItem(key, JSON.stringify(val)); }
function getSurveys() { return getData('surveys'); }
function getAnswers() { return getData('answers'); }
function saveSurveys(d) { setData('surveys', d); }
function saveAnswers(d) { setData('answers', d); }
function getPassword() { return localStorage.getItem('admin_pw') || DEFAULT_PASSWORD; }

let isAdminLoggedIn = false;

function openAdminLogin() {
  if (isAdminLoggedIn) { showPage('page-admin'); return; }
  document.getElementById('admin-password-input').value = '';
  document.getElementById('pw-error').style.display = 'none';
  document.getElementById('login-modal').classList.add('active');
  setTimeout(() => document.getElementById('admin-password-input').focus(), 100);
}
function closeModal() {
  document.getElementById('login-modal').classList.remove('active');
}
function checkPassword() {
  const input = document.getElementById('admin-password-input').value;
  if (input === getPassword()) {
    isAdminLoggedIn = true;
    closeModal();
    showPage('page-admin');
  } else {
    document.getElementById('pw-error').style.display = 'block';
    document.getElementById('admin-password-input').value = '';
    document.getElementById('admin-password-input').focus();
  }
}
function logoutAdmin() {
  isAdminLoggedIn = false;
  location.hash = '';
  showPage('page-answer');
  toast('ログアウトしました');
}
function changePassword() {
  const cur = document.getElementById('current-pw').value;
  const nw = document.getElementById('new-pw').value;
  const nwc = document.getElementById('new-pw-confirm').value;
  if (cur !== getPassword()) { toast('現在のパスワードが違います', '#e74c3c'); return; }
  if (nw.length < 4) { toast('新しいパスワードは4文字以上にしてください', '#e74c3c'); return; }
  if (nw !== nwc) { toast('新しいパスワードが一致しません', '#e74c3c'); return; }
  localStorage.setItem('admin_pw', nw);
  document.getElementById('current-pw').value = '';
  document.getElementById('new-pw').value = '';
  document.getElementById('new-pw-confirm').value = '';
  toast('✅ パスワードを変更しました！');
}

function getBaseUrl() { return location.href.split('#')[0]; }
function getSurveyUrl(surveyId) { return `${getBaseUrl()}#survey-${surveyId}`; }
function setHash(surveyId) {
  history.replaceState(null, '', surveyId ? `#survey-${surveyId}` : location.pathname + location.search);
}
function checkHash() {
  const hash = location.hash.replace('#', '');
  if (hash.startsWith('survey-')) {
    const sid = hash.replace('survey-', '');
    const s = getSurveys().find(s => String(s.id) === String(sid));
    if (s) { openForm(s.id, false); return; }
  }
  renderAnswerList();
}
function copyUrl(surveyId) {
  const url = getSurveyUrl(surveyId);
  navigator.clipboard.writeText(url).then(() => {
    toast('📋 URLをコピーしました！LINEやメールで共有できます');
  }).catch(() => {
    prompt('このURLをコピーして共有してください：', url);
  });
}

function showPage(id) {
  document.querySelectorAll('.page').forEach(p => p.classList.remove('active'));
  document.getElementById(id).classList.add('active');
  if (id === 'page-answer') renderAnswerList();
  if (id === 'page-admin') { renderAdminList(); renderResultSelect(); }
}
function goBackToList() {
  history.replaceState(null, '', location.pathname + location.search);
  showPage('page-answer');
}
function switchAdminTab(tabId, btnId) {
  document.querySelectorAll('.tab-content').forEach(t => t.classList.remove('active'));
  document.querySelectorAll('.tab-btn').forEach(b => b.classList.remove('active'));
  document.getElementById(tabId).classList.add('active');
  document.getElementById(btnId).classList.add('active');
  if (tabId === 'tab-results') renderResultSelect();
}

function toast(msg, color = '#2ecc71') {
  const t = document.getElementById('toast');
  t.textContent = msg; t.style.background = color; t.style.display = 'block';
  setTimeout(() => t.style.display = 'none', 3000);
}

let questions = [];
let editingId = null;

function addQuestion(type) {
  const q = { id: Date.now(), type, text: '', required: false, choices: type !== 'text' ? [''] : [] };
  questions.push(q);
  renderQuestionEditors();
}
function removeQuestion(id) {
  questions = questions.filter(q => q.id !== id);
  renderQuestionEditors();
}
function addChoice(qid) {
  const q = questions.find(q => q.id === qid);
  if (q) { q.choices.push(''); renderQuestionEditors(); }
}
function removeChoice(qid, idx) {
  const q = questions.find(q => q.id === qid);
  if (q && q.choices.length > 1) { q.choices.splice(idx, 1); renderQuestionEditors(); }
}
function syncQuestion(id) {
  const q = questions.find(q => q.id === id);
  if (!q) return;
  const qtEl = document.getElementById(`qt-${id}`);
  const qrEl = document.getElementById(`qr-${id}`);
  if (qtEl) q.text = qtEl.value;
  if (qrEl) q.required = qrEl.checked;
  if (q.type !== 'text') {
    q.choices = Array.from(document.querySelectorAll(`.choice-input-${id}`)).map(i => i.value);
  }
}
function renderQuestionEditors() {
  const container = document.getElementById('question-list');
  if (questions.length === 0) {
    container.innerHTML = '<p style="color:#aaa;font-size:0.9rem;margin-bottom:12px;">質問がまだありません。下のボタンで追加してください。</p>';
    return;
  }
  container.innerHTML = questions.map((q, idx) => `
    <div class="question-editor">
      <div class="q-header">
        <span>Q${idx+1}．${ q.type==='radio'?'🔘 単一選択':q.type==='checkbox'?'☑️ 複数選択':'✏️ 自由記述'}</span>
        <button class="btn btn-danger btn-sm" onclick="removeQuestion(${q.id})">削除</button>
      </div>
      <div class="form-group">
        <label>質問文</label>
        <input type="text" id="qt-${q.id}" value="${escHtml(q.text)}" placeholder="質問を入力" oninput="syncQuestion(${q.id})" />
      </div>
      <label style="display:flex;align-items:center;gap:8px;cursor:pointer;font-size:0.9rem;margin-bottom:12px;">
        <input type="checkbox" id="qr-${q.id}" ${q.required?'checked':''} onchange="syncQuestion(${q.id})" style="width:16px;height:16px;accent-color:#e74c3c;" />
        <span>必須項目にする</span>
      </label>
      ${ q.type!=='text' ? `
        <div class="choices-area">
          <label>選択肢</label>
          ${q.choices.map((c,i)=>`
            <div class="choice-row">
              <input type="text" class="choice-input-${q.id}" value="${escHtml(c)}" placeholder="選択肢 ${i+1}" oninput="syncQuestion(${q.id})" />
              <button class="btn btn-danger btn-sm" onclick="removeChoice(${q.id},${i})" ${q.choices.length<=1?'disabled':''}>✕</button>
            </div>
          `).join('')}
          <button class="btn btn-secondary btn-sm" onclick="addChoice(${q.id})" style="margin-top:4px;">＋ 選択肢を追加</button>
        </div>
      `:'' }
    </div>
  `).join('');
}

function saveSurvey() {
  const title = document.getElementById('new-survey-title').value.trim();
  if (!title) { toast('タイトルを入力してください', '#e74c3c'); return; }
  if (questions.length === 0) { toast('質問を1つ以上追加してください', '#e74c3c'); return; }
  questions.forEach(q => syncQuestion(q.id));
  const surveys = getSurveys();
  let savedId;
  if (editingId) {
    const idx = surveys.findIndex(s => s.id === editingId);
    if (idx !== -1) {
      surveys[idx] = { ...surveys[idx], title, desc: document.getElementById('new-survey-desc').value.trim(), questions: JSON.parse(JSON.stringify(questions)) };
      savedId = editingId;
    }
    editingId = null;
  } else {
    savedId = Date.now();
    surveys.push({
      id: savedId, title,
      desc: document.getElementById('new-survey-desc').value.trim(),
      questions: JSON.parse(JSON.stringify(questions)),
      createdAt: new Date().toLocaleDateString('ja-JP')
    });
  }
  saveSurveys(surveys);
  document.getElementById('new-survey-title').value = '';
  document.getElementById('new-survey-desc').value = '';
  questions = [];
  renderQuestionEditors();
  renderAdminList();
  renderResultSelect();
  toast('✅ アンケートを保存しました！');
}

function renderAdminList() {
  const surveys = getSurveys();
  const container = document.getElementById('admin-survey-list');
  if (surveys.length === 0) {
    container.innerHTML = '<p style="color:#aaa;font-size:0.9rem;">まだアンケートがありません。</p>';
    return;
  }
  const answers = getAnswers();
  container.innerHTML = surveys.map(s => {
    const count = answers.filter(a => a.surveyId === s.id).length;
    const url = getSurveyUrl(s.id);
    return `
      <div class="survey-item" style="flex-direction:column;align-items:flex-start;">
        <div style="width:100%;display:flex;justify-content:space-between;align-items:flex-start;flex-wrap:wrap;gap:8px;">
          <div>
            <h3>${escHtml(s.title)}</h3>
            <p>${escHtml(s.desc||'')}${s.desc?'&nbsp;':''} 作成日：${s.createdAt}&nbsp;
              <span class="badge badge-blue">${s.questions.length}問</span>
              <span class="badge badge-green">${count}件の回答</span>
            </p>
          </div>
          <div class="btn-group">
            <button class="btn btn-primary btn-sm" onclick="editSurvey(${s.id})">✏️ 編集</button>
            <button class="btn btn-danger btn-sm" onclick="deleteSurvey(${s.id})">🗑️ 削除</button>
          </div>
        </div>
        <div style="width:100%;margin-top:12px;background:#e8f4fd;border-radius:10px;padding:12px;">

let currentSurveyId = null;

function openForm(id, updateHash = true) {
  const s = getSurveys().find(s => s.id === id);
  if (!s) return;
  currentSurveyId = id;
  if (updateHash) setHash(id);
  document.getElementById('form-title').textContent = s.title;
  document.getElementById('respondent-name').value = '';
  const container = document.getElementById('form-questions');
  container.innerHTML = s.questions.map((q, idx) => `
    <div class="question-block">
      <div class="q-title">
        Q${idx+1}. ${escHtml(q.text)}
        ${q.required ? '<span class="required-mark">*必須</span>' : '<span style="color:#aaa;font-size:0.8rem;"> (任意)</span>'}
      </div>
      ${ q.type==='radio' ? `
        <div class="radio-group" id="q-${q.id}">
          ${q.choices.map(c=>`<label><input type="radio" name="q${q.id}" value="${escHtml(c)}" />${escHtml(c)}</label>`).join('')}
        </div>
      ` : q.type==='checkbox' ? `
        <div class="checkbox-group" id="q-${q.id}">
          ${q.choices.map(c=>`<label><input type="checkbox" name="q${q.id}" value="${escHtml(c)}" />${escHtml(c)}</label>`).join('')}
        </div>
      ` : `<textarea id="q-${q.id}" placeholder="自由に記入してください"></textarea>` }
    </div>
  `).join('');
  showPage('page-form');
}

function submitAnswer() {
  const s = getSurveys().find(s => s.id === currentSurveyId);
  if (!s) return;
  const name = document.getElementById('respondent-name').value.trim() || '匿名';
  const responses = {};
  let valid = true;
  for (const q of s.questions) {
    if (q.type === 'text') {
      const val = document.getElementById(`q-${q.id}`).value.trim();
      if (q.required && !val) { valid = false; toast(`「${q.text}」は必須です`, '#e74c3c'); break; }
      responses[q.id] = val;
    } else if (q.type === 'radio') {
      const checked = document.querySelector(`input[name="q${q.id}"]:checked`);
      if (q.required && !checked) { valid = false; toast(`「${q.text}」は必須です`, '#e74c3c'); break; }
      responses[q.id] = checked ? checked.value : '';
    } else if (q.type === 'checkbox') {
      const checked = Array.from(document.querySelectorAll(`input[name="q${q.id}"]:checked`)).map(i => i.value);
      if (q.required && checked.length === 0) { valid = false; toast(`「${q.text}」は必須です`, '#e74c3c'); break; }
      responses[q.id] = checked;
    }
  }
  if (!valid) return;
  const answers = getAnswers();
  answers.push({ id: Date.now(), surveyId: currentSurveyId, name, responses, submittedAt: new Date().toLocaleString('ja-JP') });
  saveAnswers(answers);
  toast('✅ 回答を送信しました！ありがとうございます！');
  setTimeout(() => { goBackToList(); }, 1500);
}

function renderResultSelect() {
  const surveys = getSurveys();
  const select = document.getElementById('result-survey-select');
  const current = select.value;
  select.innerHTML = '<option value="">-- アンケートを選択 --</option>' +
    surveys.map(s => `<option value="${s.id}" ${s.id==current?'selected':''}>${escHtml(s.title)}</option>`).join('');
  if (current) loadResults();
}
function loadResults() {
  const id = parseInt(document.getElementById('result-survey-select').value);
  const area = document.getElementById('result-area');
  if (!id) { area.innerHTML = ''; return; }
  const s = getSurveys().find(s => s.id === id);
  const answers = getAnswers().filter(a => a.surveyId === id);
  if (!s) { area.innerHTML = ''; return; }
  if (answers.length === 0) {
    area.innerHTML = '<div class="card"><p style="color:#aaa;">まだ回答がありません。</p></div>';
    return;
  }
  let html = `
    <div class="card">
      <h2>👥 名前別 回答一覧（${answers.length}件）</h2>
      <div style="overflow-x:auto;">
        <table class="stats-table">
          <thead><tr>
            <th>名前</th><th>回答日時</th>
            ${s.questions.map((q,i)=>`<th>Q${i+1}. ${escHtml(q.text.substring(0,12))}${q.text.length>12?'…':''}</th>`).join('')}
          </tr></thead>
          <tbody>
            ${answers.map(a=>`
              <tr>
                <td><strong>${escHtml(a.name)}</strong></td>
                <td style="white-space:nowrap;font-size:0.82rem;color:#888;">${a.submittedAt}</td>
                ${s.questions.map(q=>{
                  const v = a.responses[q.id];
                  const display = Array.isArray(v) ? v.join('、') : (v||'―');
                  return `<td>${escHtml(display)}</td>`;
                }).join('')}
              </tr>
            `).join('')}
          </tbody>
        </table>
      </div>
    </div>
  `;
  html += `<div class="card"><h2>📊 質問別 集計</h2>`;
  for (const q of s.questions) {
    html += `<div style="margin-bottom:24px;"><h3>Q. ${escHtml(q.text)}</h3>`;
    if (q.type === 'text') {
      const txts = answers.map(a => a.responses[q.id]).filter(v => v);
      html += txts.length === 0
        ? '<p style="color:#aaa;font-size:0.9rem;">回答なし</p>'
        : '<ul style="margin-left:1.2em;">' + txts.map(t=>`<li style="margin-bottom:6px;font-size:0.93rem;">${escHtml(t)}</li>`).join('') + '</ul>';
    } else {
      const tally = {};
      q.choices.forEach(c => tally[c] = 0);
      answers.forEach(a => {
        const v = a.responses[q.id];
        const vals = Array.isArray(v) ? v : [v];
        vals.forEach(val => { if (val && tally[val] !== undefined) tally[val]++; });
      });
      const total = answers.length;
      html += '<div style="overflow-x:auto;"><table class="stats-table" style="margin-top:8px;"><thead><tr><th>選択肢</th><th>票数</th><th>割合</th><th style="min-width:140px;">グラフ</th></tr></thead><tbody>';
      for (const [choice, count] of Object.entries(tally)) {
        const pct = total > 0 ? Math.round(count/total*100) : 0;
        html += `<tr><td>${escHtml(choice)}</td><td><strong>${count}</strong></td><td>${pct}%</td><td><div class="bar-wrap"><div class="bar" style="width:${pct}%"></div></div></td></tr>`;
      }
      html += '</tbody></table></div>';
    }
    html += '</div>';
  }
  html += '</div>';
  area.innerHTML = html;
}

function escHtml(s) {
  if (!s) return '';
  return String(s).replace(/&/g,'&amp;').replace(/</g,'&lt;').replace(/>/g,'&gt;').replace(/"/g,'&quot;');
}
